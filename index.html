Developer Experience

Simple, Powerful API

REST endpoints. Webhook callbacks. SDKs for every language.

dispatch-task.js
Node.js 
import FieldLayer from 'fieldlayer';

const client = new FieldLayer('fl_live_...');

// Dispatch a property inspection
const task = await client.tasks.create({
  type: 'property_inspection',
  location: {
    address: '123 Main St, Austin, TX 78701',
    access_notes: 'Gate code: 1234'
  },
  requirements: {
    photos: ['exterior', 'roof', 'foundation', 'damage'],
    data: ['condition_score', 'occupancy_status']
  },
  sla: '4h',
  webhook: 'https://api.yourapp.com/webhooks/fieldlayer'
});

// task.id → 'fl_task_abc123'
// task.status → 'dispatched'
// task.estimated_completion → '2024-01-15T14:30:00Z'
